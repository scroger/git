# GIT cheatsheet

# Config
git config --global user.name "[firstname lastname]"
git config --global user.email "[email]"

# Init or clone
git init
git clone [url]

# Work
git remote add origin [url]
git pull
git log
git status
git add [file]
git rm [file]
git commit -m "[commit message]"
git push

# Branch
git branch
git branch [branch-name]
git checkout -b [branch-name]
